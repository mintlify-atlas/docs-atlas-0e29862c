<Step title="Install react-declarative">


<Step title="Define a TypedField schema">

Create a `TypedField[]` array to describe your form. Each object specifies a `type` from `FieldType`, a `name` that maps to your data model, and optional validation callbacks.

```tsx
import { TypedField, FieldType } from "react-declarative";

interface IRegistrationData {
  firstName: string;
  lastName: string;
  email: string;
  age: string;
  agreeToTerms: boolean;
}

const fields: TypedField<IRegistrationData>[] = [
  {
    type: FieldType.Text,
    name: "firstName",
    title: "First name",
    description: "e.g. Jane",
  },
  {
    type: FieldType.Text,
    name: "lastName",
    title: "Last name",
    description: "e.g. Smith",
  },
  {
    type: FieldType.Text,
    inputType: "email",
    name: "email",
    title: "Email address",
    description: "e.g. jane@example.com",
    isInvalid({ email }) {
      const expr = /^[\w-.]+@([\w-]+\.)+[\w-]{2,4}$/g;
      if (!expr.test(email)) {
        return "Enter a valid email address";
      }
      return null;
    },
  },
  {
    type: FieldType.Text,
    name: "age",
    title: "Age",
    isInvalid({ age }) {
      const value = Number(age);
      if (!Number.isInteger(value)) {
        return "Age must be a whole number";
      }
      if (value < 1) {
        return "Age must be greater than 0";
      }
      return null;
    },
  },
  {
    type: FieldType.Checkbox,
    name: "agreeToTerms",
    title: "I agree to the terms and conditions",
  },
];
```

Every validation callback receives the entire live form state as its argument and returns an error string on failure or `null` when the field is valid.

</Step>

<Step title="Render the form with One">

Pass your `fields` array to `<One />` (or `<OneTyped />` for full TypeScript inference). Wire the `handler` prop to supply initial data and the `onChange` callback to receive updates.

```tsx
import React, { useState } from "react";
import { One, FieldType, TypedField } from "react-declarative";

// ...fields array from the previous step

export const RegistrationForm = () => {
  const [data, setData] = useState<IRegistrationData | null>(null);

  const handleChange = (newData: IRegistrationData) => {
    setData(newData);
  };

  const handleInvalidity = () => {
    setData(null);
  };

  return (
    <One<IRegistrationData>
      fields={fields}
      handler={data ?? {}}
      onChange={handleChange}
      invalidity={handleInvalidity}
    />
  );
};
```

`onChange` fires every time any field value changes. `invalidity` fires when any `isInvalid` callback returns a non-null error, which you can use to disable a submit button.

</Step>

<Step title="Observe validation in action">

Add an `isDisabled` callback to one field to see how live form state drives reactive field behaviour. Here the `email` field becomes read-only until the user supplies a first name:

```tsx
{
  type: FieldType.Text,
  inputType: "email",
  name: "email",
  title: "Email address",
  isDisabled({ firstName }) {
    return !firstName;
  },
  isInvalid({ email }) {
    const expr = /^[\w-.]+@([\w-]+\.)+[\w-]{2,4}$/g;
    if (!expr.test(email)) {
      return "Enter a valid email address";
    }
    return null;
  },
},
```

`isDisabled`, `isVisible`, and `isReadonly` all receive the same live state object as `isInvalid`, so every rule has full access to the current form data.

</Step>

</Steps>

## Complete example

The code below combines all steps into a single, self-contained component you can drop into any React project.

```tsx
import React, { useState } from "react";
import { OneTyped, FieldType, TypedField } from "react-declarative";

interface IRegistrationData {
  firstName: string;
  lastName: string;
  email: string;
  age: string;
  agreeToTerms: boolean;
}

const fields: TypedField<IRegistrationData>[] = [
  {
    type: FieldType.Text,
    name: "firstName",
    title: "First name",
    description: "e.g. Jane",
  },
  {
    type: FieldType.Text,
    name: "lastName",
    title: "Last name",
    description: "e.g. Smith",
  },
  {
    type: FieldType.Text,
    inputType: "email",
    name: "email",
    title: "Email address",
    isDisabled({ firstName }) {
      return !firstName;
    },
    isInvalid({ email }) {
      const expr = /^[\w-.]+@([\w-]+\.)+[\w-]{2,4}$/g;
      if (!expr.test(email)) {
        return "Enter a valid email address";
      }
      return null;
    },
  },
  {
    type: FieldType.Text,
    name: "age",
    title: "Age",
    isInvalid({ age }) {
      const value = Number(age);
      if (!Number.isInteger(value)) {
        return "Age must be a whole number";
      }
      if (value < 1) {
        return "Age must be greater than 0";
      }
      return null;
    },
  },
  {
    type: FieldType.Checkbox,
    name: "agreeToTerms",
    title: "I agree to the terms and conditions",
  },
];

export const RegistrationForm = () => {
  const [data, setData] = useState<IRegistrationData | null>(null);

  return (
    <OneTyped<IRegistrationData>
      fields={fields}
      handler={data ?? {}}
      onChange={(newData) => setData(newData)}
      invalidity={() => setData(null)}
    />
  );
};
```
